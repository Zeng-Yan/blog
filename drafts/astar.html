import matplotlib.pyplot as plt
import math

D = [(-1, 0), (0, -1), (0, 1), (1, 0), (-1, -1), (-1, 1), (1, -1), (1, 1)]


class AStar:
    def __init__(self) -> None:
        self.grid_map = []  # 地图
        self.frontier = []  # 边界点队列, 按照cost由小到大排序
        self.from_dict = dict()  # 坐标点对应的上一个路径点
        self.f_dict = dict()  # 坐标点对应的f花费
        self.cost_dict = dict()  # 坐标点对应的总花费
        self.h = 0  # 地图高
        self.v = 0  # 地图宽
        self.start = ()
        self.goal = ()

    def set_grid(self, grid_map):
        self.grid_map = grid_map
        self.h = len(grid_map)
        self.v = len(grid_map[0])

    def plan(self, start: tuple, goal: tuple):
        self.start = start
        self.goal = goal

        self.frontier.append(start)
        self.f_dict[start] = 0
        self.cost_dict[start] = 0  # 设置起点cost
        self.from_dict[start] = None  # 设置起点的父节点

        iter_count = 0  # 循环计数器

        while self.frontier != []:

            self.plot()
            cur_coord = self.frontier.pop(0)
            if cur_coord == goal:
                print('fin')
                break

            iter_count += 1
            if iter_count >= 1e6: # 循环超出限制, 终止
                print('out of iters')
                return

            for neighbor_coord in self.get_neighbors(cur_coord):
                neighbor_f_cost = self.f_dict[cur_coord] + 1 
                old_neighbor_f_cost = self.f_dict.get(neighbor_coord, None)
                if not old_neighbor_f_cost or neighbor_f_cost < old_neighbor_f_cost:  
                    self.f_dict[neighbor_coord] = neighbor_f_cost
                    self.cost_dict[neighbor_coord] = neighbor_f_cost + self.heuristic(neighbor_coord)
                    self.from_dict[neighbor_coord] = cur_coord
                    self.frontier.append(neighbor_coord)
                    # 把 frontier 按 cost 从小到大排序
                    self.frontier = sorted(self.frontier, key=lambda x: self.cost_dict[x])

        p = self.from_dict[goal]
        route = []
        while p != start:
            route.append(p)
            p = self.from_dict.get(p, None)
        route = [goal] + route + [start]
        print(route)
        self.plot_(route)
        return route

    def get_neighbors(self, coord):
        neighbors = []
        for i in range(4):
            sh = D[i][0]
            sv = D[i][1]
            new_coord = (coord[0] + sh, coord[1] + sv)
            if (-1 < new_coord[0] < self.h) and (-1 < new_coord[1] < self.v) and self.grid_map[new_coord[0]][new_coord[1]] < 1:
                neighbors.append(new_coord)
        return neighbors

    def g(self, coord):
        return abs(self.start[0] - coord[0]) + abs(self.start[1] - coord[1])

    def heuristic(self, coord):
        return abs(self.goal[0] - coord[0]) + abs(self.goal[1] - coord[1])

    def plot(self):
        from copy import deepcopy

        grid = deepcopy(self.grid_map)
        for i, j in self.frontier:
            grid[i][j] = 0.5

        plt.cla()
        plt.imshow(grid, cmap='gray')
        plt.pause(0.001)

    def plot_(self, route):
        plt.cla()
        plt.scatter([self.start[1], self.goal[1]], [self.start[0], self.goal[0]], marker='*')
        plt.imshow(self.grid_map, cmap='gray')
        for idx in range(len(route)-1):
            p1 = route[idx]
            p2 = route[idx+1]
            plt.plot([p1[1], p2[1]], [p1[0], p2[0]], c='r')
        plt.show()

    return abs(self.goal.x - coord.x) + abs(self.goal.y - coord.y)
