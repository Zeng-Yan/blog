[PCG] 一种地牢生成算法

Posted by Zengyan on 周三 26 六月 2024

地牢是 Rouge-like 游戏中常见的一种地形设计, 一个地牢通常包含很多房间, 房间之间通过蜿蜒曲折的通道连接起来. 在我一开始接触 Rouge-like 游戏时, 就为千奇百怪的地牢和其背后的程序生成 (Procedural generation) 算法所着迷. 这里我简单复刻一下 房间和迷宫：一个地牢生成算法 这篇文章中提出的地牢生成算法. (我没有阅读原作者的源代码, 只是按照文章中的思路进行代码实现, 因此在细节上可能会和原作者代码存在出入)

而下面的图片展示的是我使用该算法结合 Godot4 中的 Gridmap 实现的 3D 地牢原型.

生成步骤

  • 设置一个有限的地图区域
  • 在地图区域中生成多个相互不重叠的房间
  • 使用 flood-fill 策略将地图的剩下区域用通道填满
  • 找出房间和通道之间的连接位置
  • 从一个初始房间开始, 不断选择连接位置, 来将通道和房间连接起来
  • 消解地牢中的死胡同

Python 实现

代码里的连接部分有一点问题, 偶尔会有区域没能和其他区域连接上, 不过问题不大.

import matplotlib.pyplot as plt
from matplotlib.animation import ArtistAnimation
import random

D = [(-1, 0), (0, -1), (0, 1), (1, 0), (-1, -1), (-1, 1), (1, -1), (1, 1)]


class MazeMaker:
    def __init__(self, k=10) -> None:
        self.n = k*2+1
        self.maze = [[0 for i in range(self.n)] for i in range(self.n)]  # 0为黑 1为白
        self.region_count = 0    

    def _init_animation(self):
        self.fig, self.ax = plt.subplots()
        self.artists = []

    def _add_frame(self, artist):
        self.artists.append(artist)

    def _create_animation(self):
        self.artists = self.artists + [self.artists[-1] for _ in range(10)]
        ani = ArtistAnimation(self.fig, self.artists, interval=100, blit=True)
        ani.save('dungeon.gif', writer='pillow')
        plt.show()

    def make(self):
        self._init_animation()

        for _ in range(50):
            self.add_room()
        self._add_frame([plt.imshow(self.maze, cmap='gray', origin = 'lower', animated=True)])

        self.flood_fill()
        self._add_frame([plt.imshow(self.maze, cmap='gray', origin = 'lower', animated=True)])

        self.find_connects()

        self.remove_deadend()

        self._add_frame([plt.imshow(self.maze, cmap='gray', origin = 'lower', animated=True)])
        self._create_animation()

    def add_room(self):
        h = random.randint(1, 3) * 2 + 1
        v = random.randint(1, 3) * 2 + 1


        pos_x = random.randint(0, (self.n-1-h)//2) * 2 + 1
        pos_y = random.randint(0, (self.n-1-v)//2) * 2 + 1

        a = 0  # 统计当前生成区域是否已经有房间
        for i in range(pos_x, pos_x+h):
            for j in range(pos_y, pos_y+v):
                a += self.maze[i][j]

        if a == 0:
            self.region_count += 1  # 编号加一
            for i in range(pos_x, pos_x+h):
                for j in range(pos_y, pos_y+v):
                    self.maze[i][j] = self.region_count  # 标识为房间

    def flood_fill(self):
        for i in range(1, self.n, 2):
            for j in range(1, self.n, 2):
                if self.maze[i][j] == 0:  # 当前点没有房间
                    self.region_count += 1
                    self.grow_tunnel(i, j)  # DFS生成道路


    def grow_tunnel(self, pos_i, pos_j):

        if not ((0 < pos_i < self.n-1) and (0 < pos_j < self.n-1)):  # 当前点越界
            return

        if self.maze[pos_i][pos_j] > 0:  # 当前点有房间
            return

        count = 0  # 统计当前点周围墙壁个数
        for i in range(8):
            x = pos_i + D[i][0]
            y = pos_j + D[i][1]
            if self.maze[x][y] == 0:
                count += 1

        if count < 6:
            return

        self.maze[pos_i][pos_j] = self.region_count  # 标识为道路

        self._add_frame([plt.imshow(self.maze, cmap='gray', origin = 'lower', animated=True)])

        for i in range(4):  # 向四周递归
            x = pos_i + D[i][0]
            y = pos_j + D[i][1]

            self.grow_tunnel(x, y)


    def find_connects(self):
        # 寻找全部的连接点
        connects = []  # 全部的连接点
        for i in range(1, self.n-1):
            for j in range(1, self.n-1):
                if self.maze[i][j] == 0: # 该点不可通行
                    # 在纵方向上, 该点存在上下两个邻点属于不同区域
                    if self.maze[i-1][j] != 0 and self.maze[i+1][j] != 0 and self.maze[i-1][j] != self.maze[i+1][j]:
                        connects.append([i, j, self.maze[i-1][j], self.maze[i+1][j]])
                    # 在横方向上, 该点存在左右两个邻点属于不同区域
                    elif self.maze[i][j-1] != 0 and self.maze[i][j+1] != 0 and self.maze[i][j-1] != self.maze[i][j+1]:
                        connects.append([i, j, self.maze[i][j-1], self.maze[i][j+1]])

        artist = [
            plt.imshow(self.maze, cmap='gray', origin = 'lower', animated=True),
            plt.scatter([c[1] for c in connects], [c[0] for c in connects], c='r')
        ]
        self._add_frame(artist)

        # 选择单一连接点将区域联通
        merged = [1]  # 已经合并的区域
        opened = [i for i in range(2, self.region_count)]  # 待合并的区域
        old_opened = []

        doors = []
        while opened != old_opened:  # 待合并的区域没有发生变化
            old_opened = [i for i in opened]

            for idx in opened:
                related_c = self.get_region_connects(connects, [idx])  # 和idx区域 相关的连接
                related_c = self.get_region_connects(related_c, merged)  # idx 和 已合并区域的连接

                if related_c:
                    choose_c = related_c[random.randint(0, len(related_c)-1)]  # 从关联的连接中随机选取一个
                    choose_pos_x, choose_pos_y = choose_c[0], choose_c[1]
                    doors.append([choose_pos_x, choose_pos_y])
                    self.maze[choose_pos_x][choose_pos_y] = self.region_count+1  # 标记门的位置

                    merged.append(idx)
                    opened.remove(idx)

                    break

        # 随机保留部分连接点联通不同区域, 以增加回路
        for choose_c in connects:  # 随机保留部分连接点为门
            if random.random() < 0.1:
                choose_pos_x, choose_pos_y = choose_c[0], choose_c[1]

                # 避免门紧邻
                count = 0
                for i in range(4):
                    x = choose_pos_x + D[i][0]
                    y = choose_pos_y + D[i][1]
                    if self.maze[x][y] == 0:
                        count += 1

                if count >= 2:
                    doors.append([choose_pos_x, choose_pos_y])
                    self.maze[choose_pos_x][choose_pos_y] = self.region_count+1  # 标记门的位置

        artist = [
            plt.imshow(self.maze, cmap='gray', origin = 'lower'),
            plt.scatter([c[1] for c in connects], [c[0] for c in connects], c='r'),
            plt.scatter([c[1] for c in doors], [c[0] for c in doors], c='g')
        ]
        self._add_frame(artist)

    @staticmethod
    def get_region_connects(connects, region_list: list):  # 从connects中寻找相关区域的全部连接
        r = []
        for c in connects:
            if c[2] in region_list or c[3] in region_list:
                r.append(c)
        return r

    def remove_deadend(self):  # 去除死胡同
        for i in range(1, self.n, 2):
            for j in range(1, self.n, 2):
                if self.maze[i][j] > 0:
                   self.remove(i, j)

    def remove(self, pos_x, pos_y):
        if not ((0 < pos_x < self.n-1) and (0 < pos_y < self.n-1)):
            return
        if self.maze[pos_x][pos_y] == 0:
            return

        count = 0
        for i in range(4):
            x = pos_x + D[i][0]
            y = pos_y + D[i][1]
            if self.maze[x][y] == 0:
                count += 1

        if count < 3:
            return

        self.maze[pos_x][pos_y] = 0

        for i in range(4):
            x = pos_x + D[i][0]
            y = pos_y + D[i][1]

            self.remove(x, y)

class_name MazeMaker
extends Node

static var D = [[-1, 0], [0, -1], [0, 1], [1, 0], [-1, -1], [-1, 1], [1, -1], [1, 1]]

@export var k: int = 10
var n: int
var region_count: int = 0
var maze: Array = []
var rooms: Array = []  # 房间列表 [x, y, h, v]
var doors: Array = []  # 门列表 [x, y]


func make():
    rooms = []  # 重置
    doors = []  # 重置

    self.n = k*2+1

    for i in range(n):
        var line = []
        for j in range(n):
            line.append(0)
        self.maze.append(line)  # 0为黑 1为白

    self.region_count = 0

    for idx in range(5*k):
        add_room()
    flood_fill()
    find_connects()
    remove_deadend()


func add_room():
    var h = randi_range(1, 3) * 2 + 1
    var v = randi_range(1, 3) * 2 + 1

    var pos_x = randi_range(0, (n-1-h)/2) * 2 + 1
    var pos_y = randi_range(0, (n-1-v)/2) * 2 + 1

    var a = 0  # 统计当前生成区域是否已经有房间
    for i in range(pos_x, pos_x+h):
        for j in range(pos_y, pos_y+v):
            a += self.maze[i][j]

    if a == 0:
        self.region_count += 1  # 编号加一
        for i in range(pos_x, pos_x+h):
            for j in range(pos_y, pos_y+v):
                self.maze[i][j] = self.region_count  # 标识为房间
        rooms.append([pos_x, pos_y, h, v])  # 记录房间


func flood_fill():
    for i in range(1, self.n, 2):
        for j in range(1, self.n, 2):
            if self.maze[i][j] == 0:  # 当前点没有房间
                self.region_count += 1
                grow_tunnel(i, j) 


func grow_tunnel(pos_i: int, pos_j: int):
    if not (between(pos_i, 0, self.n-1) and between(pos_j, 0, self.n-1)):
        return

    if self.maze[pos_i][pos_j] > 0:  # 当前点有房间
        return

    var count = 0  # 统计当前点周围墙壁个数
    for i in range(8):
        var x = pos_i + D[i][0]
        var y = pos_j + D[i][1]
        if self.maze[x][y] == 0:
            count += 1

    if count < 6:
        return

    self.maze[pos_i][pos_j] = self.region_count  # 标识为道路

    for i in range(4):  # 向四周递归
        var x = pos_i + D[i][0]
        var y = pos_j + D[i][1]
        grow_tunnel(x, y)


func find_connects():
    # 寻找全部的连接点
    var connects = []  # 全部的连接点
    for i in range(1, self.n-1):
        for j in range(1, self.n-1):
            if self.maze[i][j] == 0: # 该点不可通行
                # 在纵方向上, 该点存在上下两个邻点属于不同区域
                if (self.maze[i-1][j] != 0) and (self.maze[i+1][j] != 0) and (self.maze[i-1][j] != self.maze[i+1][j]):
                    connects.append([i, j, self.maze[i-1][j], self.maze[i+1][j]])
                # 在横方向上, 该点存在左右两个邻点属于不同区域
                elif (self.maze[i][j-1] != 0) and (self.maze[i][j+1] != 0) and (self.maze[i][j-1] != self.maze[i][j+1]):
                    connects.append([i, j, self.maze[i][j-1], self.maze[i][j+1]])

    # 选择单一连接点将区域联通
    var merged = [1]  # 已经合并的区域
    var opened = range(2, self.region_count)  # 待合并的区域
    var old_opened = []

    while opened != old_opened:  # 待合并的区域没有发生变化
        old_opened = opened.duplicate(true)

        for idx in opened:
            var related_c = get_region_connects(connects, [idx])  # 和idx区域 相关的连接
            related_c = get_region_connects(related_c, merged)  # idx 和 已合并区域的连接

            if related_c:
                var choose_c = related_c[randi_range(0, len(related_c)-1)]  # 从关联的连接中随机选取一个
                var choose_pos_x = choose_c[0]
                var choose_pos_y = choose_c[1]
                doors.append([choose_pos_x, choose_pos_y])  # 记录门的位置
                self.maze[choose_pos_x][choose_pos_y] = self.region_count+1  # 标记门的位置

                merged.append(idx)
                opened.erase(idx)

                break

    # 随机保留部分连接点联通不同区域, 以增加回路
    for choose_c in connects:  # 随机保留部分连接点为门
        if randf() < 0.1:
            var choose_pos_x = choose_c[0]
            var choose_pos_y = choose_c[1]

            # 避免门紧邻
            var count = 0
            for i in range(4):
                var x = choose_pos_x + D[i][0]
                var y = choose_pos_y + D[i][1]
                if self.maze[x][y] == 0:
                    count += 1

            if count >= 2:
                doors.append([choose_pos_x, choose_pos_y])  # 记录门的位置
                self.maze[choose_pos_x][choose_pos_y] = self.region_count+1  # 标记门的位置


func get_region_connects(connects: Array, region_list: Array):  # 从connects中寻找相关区域的全部连接
    var r = []
    for c in connects:
        if c[2] in region_list or c[3] in region_list:
            r.append(c)
    return r


func remove_deadend():  # 去除死胡同
    for i in range(1, self.n, 2):
        for j in range(1, self.n, 2):
            if self.maze[i][j] > 0:
                remove(i, j)


func remove(pos_x: int, pos_y: int):
    if not (between(pos_x, 0, self.n-1) and between(pos_y, 0, self.n-1)):
        return
    if self.maze[pos_x][pos_y] == 0:
        return

    var count = 0
    for i in range(4):
        var x = pos_x + D[i][0]
        var y = pos_y + D[i][1]
        #print(f'{n}, {pos_i} {pos_j}, {x} {y}')
        if self.maze[x][y] == 0:
            count += 1

    if count < 3:
        return

    self.maze[pos_x][pos_y] = 0

    for i in range(4):
        var x = pos_x + D[i][0]
        var y = pos_y + D[i][1]
        remove(x, y)


func between(x, a, b):
    if x > a and x < b:
        return true
    else:
        return false

func create_from_maze():

    var maker = MazeMaker.new()
    maker.make()
    var maze = maker.maze
    var k = maker.k

    for i in range(maker.n):
        for j in range(maker.n):
            var v = maze[i][j]
            if v == 0:
                set_cell_item(Vector3(i-k, 1, j-k), 0)
            else:
                set_cell_item(Vector3(i-k, 0, j-k), 2)

    for p in maker.doors:  # 设置门
        set_cell_item(Vector3(p[0]-k, 0, p[1]-k), 1)
