import numpy as np
import math
import matplotlib.pyplot as plt
from functools import reduce


def theta2k(theta):
    #k = math.sin(theta) / math.cos(theta)
    return math.tan(math.radians(theta))


def calc_dis_p2l(x, y, a, b, c):
    """点到直线距离"""
    top = abs(a*x + b*y + c)
    bottom = math.sqrt(a**2 + b**2)
    return top/bottom


def get_line_across_p(k, x, y):
    """已知k,求过点(x,y)的直线"""
    kb = y - k*x
    return k, -1, kb


def abc2kb(a, b, c):
    """已知abc,求kb"""
    k = -a / b
    b = -c / b
    return k, b


def project_point(k, b, x, y):
    """求(x,y)在y=kx+b上的投影点"""
    x1 = (k * (y - b) + x) / (k * k + 1)
    y1 = k * x1 + b
    return x1, y1


def eval_line(points, delta):
    center_x = reduce(lambda s, x: s+x, [p[0] for p in points]) / len(points)

    return a, b, c


def center_line(points_a, points_b):
    cax = reduce(lambda s, x: s+x, [p[0] for p in points_a]) / len(points_a)
    cay = reduce(lambda s, y: s+y, [p[1] for p in points_a]) / len(points_a)

    cbx = reduce(lambda s, x: s+x, [p[0] for p in points_b]) / len(points_b)
    cby = reduce(lambda s, y: s+y, [p[1] for p in points_b]) / len(points_b)

    cx = (cax + cbx) / 2
    cy = (cay + cby) / 2

    k = (cby - cay) / (cbx - cax)
    k = -1 / k
    a, b, c = get_line_across_p(k, cx, cy)
    min_x, max_x = min([p[0] for p in points_a]), max([p[1] for p in points_b])
    y1, y2 = - (a * min_x + c) / b, - (a * max_x + c) / b
    plt.plot([min_x, max_x], [y1, y2])

    k, b = abc2kb(a, b, c)
    pj_a = [project_point(k, b, p[0], p[1]) for p in points_a]
    idxa = list(range(len(pj_a)))
    idxa.sort(key=lambda i: pj_a[i][0])
    points_a = [points_a[i] for i in idxa]

    pj_b = [project_point(k, b, p[0], p[1]) for p in points_b]
    idxb = list(range(len(pj_b)))
    idxb.sort(key=lambda i: pj_b[i][0])
    points_b = [points_b[i] for i in idxb]

    for i in range(len(points_a)):
        plt.plot([points_a[i][0], points_b[i][0]], [points_a[i][1], points_b[i][1]])

    return

# 计算经纬度距离
from geopy.distance import geodesic
import math


def haversine(p1, p2):
    """
    Calculate the great circle distance between two points 
    on the earth (specified in decimal degrees)
    """
    # 将十进制度数转换为弧度
    lat1, lon1, lat2, lon2 = map(math.radians, [p1[1], p1[0], p2[1], p2[0]])

    # Haversine formula
    dlat = lat2 - lat1
    dlon = lon2 - lon1
    a = math.sin(dlat/2)**2 + math.cos(lat1) * math.cos(lat2) * math.sin(dlon/2)**2
    c = 2 * math.asin(math.sqrt(a))

    # 地球半径（平均半径，单位：公里）
    radius_of_earth_km = 6371.0

    # 计算出的实际距离
    distance = radius_of_earth_km * c * 1000
    return distance

d1 = geodesic((p1[1], p1[0]), (p2[1], p2[0])).km
