栅格地图 (一) 直线光栅化

Posted by Zengyan on 周二 27 六月 2023

在工作中频繁涉及对栅格地图的处理, 所以写一些笔记来进行归纳总结.

在栅格地图中做直线的光栅化主要有两个目的, 一个是标识区域, 一个是用于碰撞检测.

import numpy as np
import matplotlib.pyplot as plt
from typing import Optional, Tuple
from loguru import logger
# Rasterization


def calc_l2_d(p1, p2):
    p1 = np.array(p1)
    p2 = np.array(p2)
    distance = np.sqrt(np.sum(np.square(p2 - p1)))
    return distance


def flood_fill(mat: np.ndarray, p: Tuple[int, int], mat_vld=0) -> np.ndarray:
    rows, cols = mat.shape
    queue = [p]
    orient = [0, 1, 0, -1]

    mask = np.zeros_like(mat)
    while queue:
        px, py = queue[0]
        queue = queue[1:]
        for i in range(4):
            r, c = px + orient[i], py + orient[-i-1]
            if 0 <= r < rows and 0 <= c < cols:
                if mat[r, c] == mat_vld and mask[r, c] == 0:
                    mask[r, c] = 1
                    queue.append((r, c))
    return mask


def discrete_bold_raycast(uv1: Tuple[int, int], uv2: Tuple[int, int], step=0.2) -> bool:
    (x1, y1), (x2, y2) = uv1, uv2
    a, b, c = y2 - y1, x1 - x2, x2 * y1 - x1 * y2
    logger.debug(f'Raycast points: {x1, y1}, {x2, y2}.')
    logger.debug(f'Raycast params: {a}x {b}y {c}.')
    if b == 0:
        y = np.arange(y1, y2) if a > 0 else np.arange(y1, y2, -1)
        x = np.ones_like(y) * x1
    else:
        sample_x = np.arange(x1, x2, step) if b < 0 else np.arange(x1, x2, -step)  # x 轴密布, y轴会有稀疏
        spy = (- (a * sample_x + c) / b)
        spy_f = np.floor(spy).astype(int)
        spy_c = np.ceil(spy).astype(int)
        sample_x = sample_x.astype(int)

        sample_y = np.arange(y1, y2, step) if a > 0 else np.arange(y1, y2, -step)  # y轴密布, x轴会有稀疏
        spx = (-(b * sample_y + c) / a)
        spx_f = np.floor(spx).astype(int)
        spx_c = np.ceil(spx).astype(int)
        sample_y = sample_y.astype(int)

        x = np.concatenate([sample_x, sample_x, spx_f, spx_c])
        y = np.concatenate([spy_f, spy_c, sample_y, sample_y])

    return x, y


def discrete_slim_raycast(uv1: Tuple[int, int], uv2: Tuple[int, int]):
    (x1, y1), (x2, y2) = uv1, uv2
    a, b, c = y2 - y1, x1 - x2, x2 * y1 - x1 * y2
    logger.debug(f'Raycast params: {x1, y1}, {x2, y2}.')
    logger.debug(f'Raycast params: {a}x {b}y {c}.')
    if b == 0:  # 垂直
        index_y = np.arange(y1, y2) if a > 0 else np.arange(y1, y2, -1)
        index_x = np.ones_like(index_y) * x1
    else:
        k = a / b
        absk = abs(k)  # 斜率的绝对值
        if absk > 1:  #斜率高 上下走向
            index_y = np.arange(y1, y2+1, 1) if a > 0 else np.arange(y2, y1+1, 1) # 因为点始终是方格的中点, 所以判断会有些反直觉, 
            checkpoints_x = (-(b * index_y + c) / a)
            index_x = np.round(checkpoints_x).astype(int)
        elif absk < 1:  # 斜率低 左右走向
            index_x = np.arange(x1, x2+1, 1) if b < 0 else np.arange(x2, x1+1, 1)
            checkpoints_y = (- (a * index_x + c) / b)
            index_y = np.round(checkpoints_y).astype(int)
        else:  # k=1
            index_x = np.arange(x1, x2+1) if b < 0 else np.arange(x1, x2-1, -1)
            index_y = np.arange(y1, y2+1) if a > 0 else np.arange(y1, y2-1, -1)
    return index_x, index_y


def create_circle_mask(shape: Tuple[int, int], ctlp: Tuple[float, float], radius: float, cell_length=1., dfunc=calc_l2_d) -> np.ndarray:

    orient = [0, 1, 0, -1]

    x, y = ctlp[0], ctlp[1]
    mask = np.zeros(shape)
    mask[int(x), int(y)] = 1
    rows, cols = mask.shape
    queue = [(x, y)]
    while queue:
        px, py = queue[0]
        queue = queue[1:]
        for i in range(4):
            r, c = px + orient[i], py + orient[-i-1]
            if 0 <= r < rows and 0 <= c < cols:  # 越界判断
                if mask[r, c] == 0:  # 未遍历
                    distance = dfunc((r, c), (x, y))
                    if (distance - radius) < cell_length:
                        mask[r, c] = 1
                        queue.append((r, c))

    return mask


def create_poly_mask(shape: Tuple[int, int], points, inner: Optional[Tuple[int, int]]=None, typ='bold'):
    match typ:
        case 'bold':
            raycast = discrete_bold_raycast
        case 'slim':
            raycast = discrete_slim_raycast
        case _:
            logger.warning(f'Unsupported type: {typ}')
            raycast = discrete_bold_raycast

    mask = np.zeros(shape)
    for i in range(len(points)-1):
        p1, p2 = points[i], points[i+1]
        x, y = raycast(p1, p2)
        mask[x, y] = 1
    x, y = raycast(points[-1], points[0])
    mask[x, y] = 1

    if inner:
        inner_mask = flood_fill(mask, inner)
        mask = inner_mask + mask

    return mask


def is_p_in_mask(p, mask):
    return True if mask[p[0], p[1]] == 1 else False


def draw_grid(ax, shape, color='white'):
    rows, cols = shape
    for i in range(rows):
        r = np.arange(cols+1) - 0.5
        c = np.ones(cols+1)*i - 0.5
        ax.plot(r, c, c=color)
    for j in range(cols):
        r = np.ones(rows+1)*j - 0.5
        c = np.arange(rows+1) - 0.5
        ax.plot(r, c, c=color)
    return ax


def draw_center(ax, shape, color='blue'):
    rows, cols = shape
    for i in range(rows):
        for j in range(cols):
            ax.scatter(i, j, marker='.', c=color)
    return ax

class RasterImgPainter:

    COLORS = {
        'red': (235, 59, 90),
        'yellow': (247, 183, 49),
        'green': (32, 191, 107),
        'blue': (56, 103, 214),
        'purple': (136, 84, 208),
        'dark': (75, 101, 132),
        'light': (165, 177, 194)
    }

    def __init__(self, rows=1, cols=1) -> None:
        # plt.clf()

        assert rows+cols >=2, logger.error('Invaild shape for subplots.')

        self.rows, self.cols = rows, cols
        self.fig, self.axes = plt.subplots(nrows=rows, ncols=cols)
        self.zorder = np.ones((rows, cols))


    def get_ax(self, ax_loc=(0, 0)):
        if (self.rows + self.cols) == 2:
            return self.axes
        else:
            return self.axes[ax_loc[0], ax_loc[1]]

    def add_mask(self, mask: np.ndarray, ax_loc=(0, 0), rgb=(32,191,107), label='mask'):
        axis = self.get_ax(ax_loc)
        rgb = np.array(rgb) / 255

        mask = mask.T
        color_mask = np.ones((mask.shape[0], mask.shape[1], 3))
        color_mask = color_mask * rgb
        color_mask[mask==0] = np.array((1,1,1))

        self.zorder[ax_loc[0], ax_loc[1]] += 1
        axis.imshow(color_mask, zorder=self.zorder[ax_loc[0], ax_loc[1]])
        axis.hist([], color=rgb, label=label)

    def add_line(self, x, y, ax_loc=(0, 0), rgb=(32,191,107), label='line'):
        axis = self.get_ax(ax_loc)
        rgb = np.array(rgb) / 255

        self.zorder[ax_loc[0], ax_loc[1]] += 1
        axis.plot(x, y, c=rgb, linewidth=3, label=label, zorder=self.zorder[ax_loc[0], ax_loc[1]])

    def set_grid(self, shape, ax_loc=(0, 0), rgb=(255,255,255), with_grid=True, with_grid_center=False):

        shape = (int(shape[1]), int(shape[0]))  
        axis = self.get_ax(ax_loc)
        rgb = np.array(rgb) / 255

        # locs = axis.get_xticks()
        if with_grid:
            axis = draw_grid(axis, shape, rgb)
        if with_grid_center:
            axis = draw_center(axis, shape, rgb)

    def set_title(self, title, ax_loc=(0, 0)):
        axis = self.get_ax(ax_loc)
        axis.set_title(title)

    def show(self):
        for i in range(self.rows):
            for j in range(self.cols):
                axis = self.get_ax((i, j))
                axis.legend()
        plt.show()
