栅格地图 (二) DEM高程地图处理

Posted by Zengyan on 周二 27 六月 2023

要使用 gdal-python 来读取和处理 tif 格式的高程地图, 其核心在于理解 gdal 中的 GetGeoTransform 方法并使用它来进行坐标转换. 下面我绘制了一张图片来帮助理解 GetGeoTransform 方法返回结果的含义: 对于读取的方形区域图像, \(GT[0]\) 和 \(GT[3]\) 标识了区域的左上角顶点的经纬度, \(GT[1]\) 和 \(GT[5]\) 表示该图像每个像素对应经纬度的大小, 以 \(\theta\) 表示区域相对正北方向的偏移角度, \(GT[2]\) 和 \(GT[4]\) 表示 \(-\tanh\theta\) 的值.

这里使用到了第一部分光栅化 src.raster_utils 中的部分函数.

import numpy as np
from loguru import logger
from typing import Any, Tuple, List, Dict, Optional

from src.raster_utils import (
    flood_fill, create_circle_mask, create_poly_mask
)

CONSTANTS_RADIUS_OF_EARTH = 6371000.     # meters (m)


def geo_to_xy(lon, lat, ref_lon=0., ref_lat=0.) -> Tuple[float, float]:
    """
    input GPS ($lon, $lat) in degrees (longitude, latitude) and Reference GPS ($ref_lon, $ref_lat) in degrees, output (x, y) in meters(m) (North, East).
    """

    lat_rad = np.radians(lat)
    lon_rad = np.radians(lon)
    ref_lat_rad = np.radians(ref_lat)
    ref_lon_rad = np.radians(ref_lon)

    sin_lat = np.sin(lat_rad)
    cos_lat = np.cos(lat_rad)
    ref_sin_lat = np.sin(ref_lat_rad)
    ref_cos_lat = np.cos(ref_lat_rad)

    cos_d_lon = np.cos(lon_rad - ref_lon_rad)

    arg = np.clip(ref_sin_lat * sin_lat + ref_cos_lat * cos_lat * cos_d_lon, -1.0, 1.0)
    c = np.arccos(arg)

    k = 1.0
    if abs(c) > 0:
        k = (c / np.sin(c))

    x = float(k * (ref_cos_lat * sin_lat - ref_sin_lat * cos_lat * cos_d_lon) * CONSTANTS_RADIUS_OF_EARTH)
    y = float(k * cos_lat * np.sin(lon_rad - ref_lon_rad) * CONSTANTS_RADIUS_OF_EARTH)

    return x, y


def xy_to_geo(x, y, ref_lon=0., ref_lat=0.) -> Tuple[float, float]:
    """
    input ($x, $y) in meters(m) (North, East) and Reference GPS ($ref_lon, $ref_lat) in degrees, output GPS.
    """
    x_rad = float(x) / CONSTANTS_RADIUS_OF_EARTH
    y_rad = float(y) / CONSTANTS_RADIUS_OF_EARTH
    c = np.sqrt(x_rad * x_rad + y_rad * y_rad)

    ref_lat_rad = np.radians(ref_lat)
    ref_lon_rad = np.radians(ref_lon)

    ref_sin_lat = np.sin(ref_lat_rad)
    ref_cos_lat = np.cos(ref_lat_rad)

    if abs(c) > 0:
        sin_c = np.sin(c)
        cos_c = np.cos(c)

        lat_rad = np.asin(cos_c * ref_sin_lat + (x_rad * sin_c * ref_cos_lat) / c)
        lon_rad = (ref_lon_rad + np.atan2(y_rad * sin_c, c * ref_cos_lat * cos_c - x_rad * ref_sin_lat * sin_c))

        lat = np.degrees(lat_rad)
        lon = np.degrees(lon_rad)

    else:
        lat = np.degrees(ref_lat)
        lon = np.degrees(ref_lon)

    return lon, lat


def calc_geo_distance(p1, p2) -> np.ndarray:
    """
    given (longitude, latitude, altitude) or (longitude, latitude) format point $p1 and $p2,
    calcluate distance between $p1 and $p2.
    """
    c1 = geo_to_xy(p1[0], p1[1])
    c2 = geo_to_xy(p2[0], p2[1])
    if len(p1) == len(p2) == 2:
        p1 = np.array([c1[0], c1[1]])
        p2 = np.array([c2[0], c2[1]])
        distance = np.sqrt(np.sum(np.square(p2 - p1)))
    elif len(p1) == len(p2) == 3:
        p1 = np.array([c1[0], c1[1], p1[2]])
        p2 = np.array([c2[0], c2[1], p2[2]])
        distance = np.sqrt(np.sum(np.square(p2 - p1)))
    else:
        raise ValueError(f'Wrong dimensions of input: expect same dimensions of 2 or 3, but given {len(p1)} and {len(p2)}')
    return distance


def simple_down_sampling(mat: np.ndarray, scale: int) -> np.ndarray:
    rows, cols = mat.shape
    idx_r = np.arange(0, rows, scale)
    idx_c = np.arange(0, cols, scale)
    return mat[idx_r][:, idx_c]


def simple_edge_detect(mat: np.ndarray, thresh: Any) -> np.ndarray:
    shift_r = np.roll(mat, 1, axis=0)
    shift_c = np.roll(mat, 1, axis=1)

    detect = np.abs(mat - shift_r) + np.abs(mat - shift_c)
    detect[detect < thresh] = 0
    detect[detect > 0] = 1

    return detect


def scale_height(mat: np.ndarray, low: Any, high: Any) -> np.ndarray:
    h_max, h_min = np.max(mat), np.min(mat)
    # print(h_max, h_min)
    return (mat - h_min) / (h_max - h_min) * (high-low) + low


class BaseDEMMap:
    def __init__(self, dataset, band: int = 1):
        # print(dataset.GetMetadata())  # 元数据
        nx, ny = dataset.RasterXSize, dataset.RasterYSize  # x及y方向的像素计数
        bands = dataset.RasterCount  # 波段数
        band = dataset.GetRasterBand(band)  # 获取特定波段

        # 获取地图坐标范围
        geo_transform = dataset.GetGeoTransform()  # 地理坐标信息
        logger.debug(f'GEO METADATA: {dataset.GetMetadata()}')
        logger.debug(f'GEO TRANSFORM: {dataset.GetGeoTransform()}')
        x_min, y_min = geo_transform[0], geo_transform[3]  # 左上角x，y的坐标
        x_max = x_min + nx * geo_transform[1] + ny * geo_transform[2]
        y_max = y_min + nx * geo_transform[4] + ny * geo_transform[5]

        #  (x_min, y_max)* -- (x_max, y_max)
        #        |                  |
        #  (x_min, y_min)  -- (x_max, y_min)*

        lon_min, lon_max = x_min, x_max
        lat_max, lat_min = y_min, y_max
        logger.debug(f'GEO RANGE: ({lon_min}, {lat_min}), ({lon_max}, {lat_max}), {nx} X {ny}')

        # 坐标及高程数据
        x, y = np.linspace(x_min, x_max, nx), np.linspace(y_min, y_max, ny)
        self.X, self.Y = np.meshgrid(x, y)
        self.Z: np.ndarray = band.ReadAsArray(0, 0, nx, ny)  # type: ignore # 这一段就是讲数据的x，y，z化作numpy矩阵

        self.origin_X, self.origin_Y, self.origin_Z = np.copy(self.X), np.copy(self.Y), np.copy(self.Z)

        # 其他
        self._obstacles = dict()

        logger.debug(f'GEO cell length: {self.get_cell_length()}')

    def aux_operate(self, func: Any, coord: str='xyz', **kwargs):
        """
        apply function $func with its params to chosen coordinates given by string $coord.
        """
        affine = {'x': self.X, 'y': self.Y, 'z': self.Z}
        self.X, self.Y, self.Z = [func(affine[k], **kwargs) if k in coord else affine[k] for k in ['x', 'y', 'z']]

    def coord_to_geo(self, x: Any, y: Any, with_h=True) -> Tuple[float, float, float] | Tuple[float, float]:
        """
        convert (row, col) format coordinate ($x, $y) into (longitude, latitude, altitude).
        """
        if with_h:
            return self.X[x, y], self.Y[x, y], self.Z[x, y]  # type: ignore # 经度，纬度, 海拔
        else:
            return self.X[x, y], self.Y[x, y]  # type: ignore

    def geo_to_coord(self, longitude: Any, latitude: Any) -> Tuple[int, int]:
        """
        convert ($longitude, $latitude) into (row, col) format coordinate.
        """
        xs, xe, ys, ye = self.X[0, 0], self.X[-1, -1], self.Y[0, 0], self.Y[-1, -1]  #
        rows, cols = self.Z.shape
        x = ((longitude - xs) / (xe - xs)) * cols
        y = ((latitude - ys) / (ye - ys)) * rows
        return int(y), int(x)

    def _geo_to_coord_on_origin(self, longitude: Any, latitude: Any) -> Tuple[int, int]:
        """
        convert ($longitude, $latitude) into (row, col) format coordinate according to the original map.
        """
        xs, xe, ys, ye = self.origin_X[0, 0], self.origin_X[-1, -1], self.origin_Y[0, 0], self.origin_Y[-1, -1]  #
        rows, cols = self.origin_X.shape
        x = ((longitude - xs) / (xe - xs)) * cols
        y = ((latitude - ys) / (ye - ys)) * rows
        return int(y), int(x)

    def legitimize_point(self, p: List) -> List:
        """
        check and relocate a given geo location $p to make sure it located in map region.
        """

        rows, cols = self.Z.shape
        lon_min, lon_max = self.X[0, 0], self.X[rows-1, cols-1]
        lat_max, lat_min = self.Y[0, 0], self.Y[rows-1, cols-1]

        lon, lat = p[0], p[1]
        lon = max(lon, lon_min)
        lon = min(lon, lon_max)

        lat = max(lat, lat_min)
        lat = min(lat, lat_max)

        old_p = np.array(p[0:2])
        new_p = np.array([lon, lat])

        if np.sum(old_p - new_p) != 0:
            logger.warning(f'Point out of map, relocate point {old_p} to {new_p}.')

        return p

    def set_area(self, top_left_pos, bottom_right_pos) -> None:
        """
        clip a rectangle area from the original map.
        """
        top_left_pos = self.legitimize_point(top_left_pos)
        bottom_right_pos = self.legitimize_point(bottom_right_pos)

        tf_x, tf_y = self._geo_to_coord_on_origin(top_left_pos[0], top_left_pos[1])
        br_x, br_y = self._geo_to_coord_on_origin(bottom_right_pos[0], bottom_right_pos[1])

        region = np.s_[tf_x:br_x, tf_y:br_y]
        self.X, self.Y, self.Z = self.origin_X[region], self.origin_Y[region], self.origin_Z[region]

        return None

    def get_cell_length(self):
        return calc_geo_distance(self.coord_to_geo(0, 0, with_h=False), self.coord_to_geo(0, 1, with_h=False))


class DEMMap(BaseDEMMap):

    def create_circle_mask(self, ctlp: Tuple[float, float], radius: float) -> np.ndarray:
        ctlp = self.legitimize_point(ctlp)

        shape = self.Z.shape

        def geo_distance_with_coord(p1, p2):
            geo1 = self.coord_to_geo(p1[0], p1[1], with_h=False)
            geo2 = self.coord_to_geo(p2[0], p2[1], with_h=False)
            return calc_geo_distance(geo1, geo2)

        cell_length = calc_geo_distance(self.coord_to_geo(0, 0, with_h=False), self.coord_to_geo(0, 1, with_h=False))
        ctlp = self.geo_to_coord(ctlp[0], ctlp[1])
        mask = create_circle_mask(shape, ctlp, radius, cell_length, geo_distance_with_coord)

        return mask

    def create_poly_mask(self, p_list: List, ctlp=None, typ='slim') -> np.ndarray:
        p_list = [self.legitimize_point(p) for p in p_list]
        ctlp = self.legitimize_point(ctlp) if ctlp else None

        shape = self.Z.shape
        points = [self.geo_to_coord(p[0], p[1]) for p in p_list]
        mask = create_poly_mask(shape, points, self.geo_to_coord(*ctlp), typ)

        return mask

    def create_h_mask(self, h_threshold):
        mask = np.zeros_like(self.Z)
        mask[self.Z >= h_threshold] = 1
        return mask

    def wrap_path_to_dict(self, route: List[Tuple[int, int]]) -> Dict[str, Dict]:
        """
        return a dict describing a path by giving a list composed of (x, y). and it will be like:
            {'index':{'lng': longitude, 'lat': latitude, 'alt': altitude}}
        """
        path_dict = {}
        for idx, (x, y) in enumerate(route):
            longitude, latitude, altitude = self.coord_to_geo(x, y, with_h=True)  # type: ignore
            path_dict[f'{idx}'] = {'lng': float(longitude), 'lat': float(latitude), 'alt': float(altitude)}  # 经纬高
            # path_dict[f'{idx}'] = [longitude, latitude, altitude]  # 经纬高
        return path_dict

    def get_binary_image(self, height_diff_thresh = 20) -> np.ndarray:
        # 预处理
        self.aux_operate(simple_down_sampling, 'xyz', scale=13)  # 下采样
        self.aux_operate(scale_height, 'z', low=100, high=200)  # 高度缩放

        grid = simple_edge_detect(self.Z, height_diff_thresh)  # 坡度检测
        grid_binary = 1 - flood_fill(grid, (10, 10))  # 洪水去噪
        return grid_binary

    def plot(self, route: Optional[List[Tuple[int, int]]] = None) -> None:
        """
        plot a 3D surface of this DEM image with a path 'route' on it.
        """
        from matplotlib import cm
        from matplotlib.colors import LightSource
        import matplotlib.pyplot as plt

        # plot surface
        # plt.clf()
        fig, ax = plt.subplots(subplot_kw={'projection': '3d'}, figsize=(12, 10))
        ls = LightSource(270, 20)  # 设置你可视化数据的色带
        rgb = ls.shade(self.Z, cmap=cm.gist_earth, vert_exag=0.1, blend_mode='soft')  # type: ignore
        surf = ax.plot_surface(self.X, self.Y, self.Z, rstride=1, cstride=1, facecolors=rgb, linewidth=0, antialiased=False, shade=False, zorder=10)  # type: ignore
        # surf = ax.plot_wireframe(X, Y, Z, antialiased=False)

        # trick
        z_max = (np.max(self.Z) - np.min(self.Z)) * 7 + np.min(self.Z)
        px, py = self.coord_to_geo(0, 0, with_h=False)  # type: ignore
        ax.plot3D([px], [py], [z_max], 'gray')  # type: ignore

        # plot route
        if route:
            path_x = [self.X[x, y] for x, y in route]
            path_y = [self.Y[x, y] for x, y in route]
            path_z = [self.Z[x, y] for x, y in route]
            ax.plot(path_x, path_y, path_z, 'white', zorder=100)

        # plt.savefig('surface.jpg')
        plt.show()
