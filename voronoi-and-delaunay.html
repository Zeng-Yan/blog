Voronoi 图和 Delaunay 三角剖分

Voronoi 图和 Delaunay 三角剖分均是图形处理, 计算几何, 地理信息系统等多个领域中的重要技术.

Delaunay 三角剖分是一种点集的三角剖分标准, 它满足两个核心特性:

  • 最大化最小角特性: 在点集的所有可能的三角剖分中, 该剖分形成三角形的最小内角最大. 这有助于避免出现长而窄的三角形.
  • 空圆特性: 任意一个三角形的外接圆内不得包含点集中其他的点, 这种形式的剖分产生的最小内角最大.

Voronoi 图和 Delaunay 三角剖分是对偶的, 两者可以通过数学变换相互转换.
具体而言, Delaunay 三角剖分的顶点可以看作是 Voronoi 图的多边形中心, 而 Voronoi 多边形的边界则由 Delaunay 三角形的垂直平分线构成.

实现原理

Bowyer-Watson 算法是一种用于动态构建 Delaunay 三角剖分的有效方法, 其核心是新插入一个顶点到现有的三角剖分中的更新策略:

  • 定位受影响三角形: 找到所有外接圆覆盖新插入顶点的三角形, 这些三角形构成了需要重新构型的区域.
  • 边缘检测与重构: 识别上述三角形集合的公共边, 去除后剩下的边将与新顶点一起构成新的三角形.

为了能进行这一个更新策略, 需要构造一个初始环境, 即创建一个能够将全部顶点包含在内部的超级三角形作为初始剖分.
随后按上诉更新策略将点集中的点逐个插入, 最后删除和超级三角形相关的三角形, 如此就得到了最终的三角剖分结果.

而在构造了 Delaunay 三角剖分后, 三角形的外心就是 Voronoi 图多边形的顶点, 三角形的外心和中垂线便构成 Voronoi 图多边形的边.

import matplotlib.pyplot as plt
from __future__ import annotations


class Triangle:
    def __init__(self, p1: tuple, p2: tuple, p3: tuple) -> None:
        self.v1 = p1
        self.v2 = p2
        self.v3 = p3

        self._calc_circumcircle()

    def get_vertexes(self) -> list[tuple]:
        """返回三角形顶点
        """
        return [self.v1, self.v2, self.v3]

    def get_circumcircle(self) -> list:
        """返回外接圆 [x坐标, y坐标, 半径的平方]
        """
        return [self.x, self.y, self.r2]

    def is_in_circumcircle(self, p: tuple) -> bool:
        """判断点p是否在三角形的外接圆内部
        """
        d2 = (self.x-p[0])**2+(self.y-p[1])**2  # 距离的平方
        if d2 <= self.r2:
            return True
        else:
            return False

    def _calc_circumcircle(self) -> None:
        """计算三角形的外接圆
        """
        x1, y1 = self.v1 
        x2, y2 = self.v2
        x3, y3 = self.v3

        a = x1*(y2-y3) - y1*(x2-x3) + x2*y3 - x3*y2
        b = (x1**2+y1**2)*(y3-y2) + (x2**2+y2**2)*(y1-y3) + (x3**2+y3**2)*(y2-y1)
        c = (x1**2+y1**2)*(x2-x3) + (x2**2+y2**2)*(x3-x1) + (x3**2+y3**2)*(x1-x2)
        d = (x1**2+y1**2)*(x3*y2-x2*y3) + (x2**2+y2**2)*(x1*y3-x3*y1) + (x3**2+y3**2)*(x2*y1-x1*y2)

        if a == 0:
            raise ValueError(f'三角形不存在: {self.get_vertexes()}')

        self.x = -0.5*b/(a + 1e-8)
        self.y = -0.5*c/(a + 1e-8)
        self.r2 = (b**2+c**2-4*a*d)/(4*a**2 + 1e-8)

    def is_adjacent(self, t: Triangle) -> bool:
        """和另一个三角形t是否有相邻边
        """
        vertexes1 = self.get_vertexes()
        vertexes2 = t.get_vertexes()

        count = 0
        for v in vertexes1:
            if v in vertexes2:
                count += 1

        if count == 2:  # 有两个顶点相同
            return True
        else:
            return False


class Delaunay:
    def __init__(self) -> None:
        self.vertexes: list[tuple] = []
        self.triangles: list[Triangle] = []
        self.super_triangle: Triangle = None

    def _create_super_triangle(self) -> None:
        """创建一个内部包含所有顶点的超级三角形
        """
        vertexes_x = [v[0] for v in self.vertexes]
        vertexes_y = [v[1] for v in self.vertexes]

        min_x, max_x = min(vertexes_x)-1, max(vertexes_x)+1
        min_y, max_y = min(vertexes_y)-1, max(vertexes_y)+1

        x = (max_x -min_x)/2
        y = max_y - min_y
        origin = (min_x + x, min_y)

        p1 = (origin[0], origin[1]+2*y)
        p2 = (origin[0]-2*x, origin[1])
        p3 = (origin[0]+2*x, origin[1])

        super_triangle = Triangle(p1, p2, p3)
        self.triangles.append(super_triangle)
        self.super_triangle = super_triangle

    @staticmethod
    def _is_same_edge(edge1, edge2):
        """判断两条边是否相同
        """
        if edge1[0] == edge2[0] and edge1[1] == edge2[1]:  # 两条边是一样的
            return True
        elif edge1[1] == edge2[0] and edge1[0] == edge2[1]:  # 两条边是一样的
            return True
        else:
            return False

    @staticmethod
    def _get_dedup_edges(triangles: list[Triangle]) -> list:
        """获取列表中三角形的全部边, 并且将存在重合的边排除在外
        """
        edges = []  # 全部遍历过的边
        dedup_edges = []  # 将重合边排除在外的边, 不仅仅是不重合的边
        for triangle in triangles:
            vertexes = triangle.get_vertexes()
            for idx in range(3):  # 遍历三角形的全部边
                edge = (vertexes[idx-1], vertexes[idx])

                flag = False
                for prev_edge in edges:  # 和集合中的全部边比较
                    if Delaunay._is_same_edge(prev_edge, edge):
                        flag = True
                        dedup_edges.remove(prev_edge)  # 排除该重合边
                        break
                edges.append(edge)
                if not flag:  # 没有重合的边
                    dedup_edges.append(edge)

        return dedup_edges

    def _add_vertex(self, p: tuple) -> None:
        """逐步添加点p到现有的三角网中
        """
        related_tri = []  # p在其外接圆内部的三角形
        irrelated_tri = []  # p在其外接圆外部的三角形
        for triangle in self.triangles:
            if triangle.is_in_circumcircle(p):
                related_tri.append(triangle)
            else:
                irrelated_tri.append(triangle)
        self.triangles = irrelated_tri

        related_edges = self._get_dedup_edges(related_tri)  # 获取相关三角形的全部边 并排除重合的边
        for edge in related_edges:  # 将这些边和新加入的点组成三角形
            self.triangles.append(Triangle(edge[0], edge[1], p))

    @staticmethod
    def _exclude_triangles_with_vertexes(triangles: list[Triangle], vertexes: list[tuple]):
        """从三角形列表triangles中排除与顶点列表vertexes相关的三角形
        """
        new_triangles = []
        for triangle in triangles:
            flag = False
            for v in triangle.get_vertexes():
                if v in vertexes:
                    flag = True
                    break
            if not flag:
                new_triangles.append(triangle)
        return new_triangles

    def get_triangulation(self, vertexes: list[tuple]) -> list[Triangle]:
        """Get Delaunay triangulation of given vertexes.

        Args: vertexes: List[Tuple], list of vertexes
        Returns: triangulation: List[Triangle], triangulation
        """
        self.vertexes = vertexes

        self._create_super_triangle()  # 构建超级三角形
        for v in self.vertexes:  # 逐个点加入
            self._add_vertex(v)

        super_triangle_vertexes = self.super_triangle.get_vertexes()
        triangulation = self._exclude_triangles_with_vertexes(self.triangles, super_triangle_vertexes)  # 将包含超级三角形顶点的三角形排除
        self.triangles = triangulation
        return triangulation

    @staticmethod
    def _get_adjacent_triangles(triangle: Triangle, triangles: list[Triangle]) -> list[Triangle]:
        """从triangles获取三角形triangle相邻的三角形
        """
        adjust_triangles = []
        for t in triangles:
            if t.is_adjacent(triangle):
                adjust_triangles.append(t)
        return adjust_triangles

    def to_voronoi(self):
        passed_triangles = []

        edges = []
        for triangle in self.triangles:
            adjacent_triangles = self._get_adjacent_triangles(triangle, self.triangles)
            for t in adjacent_triangles:
                if t not in passed_triangles:
                    edges.append(((triangle.x, triangle.y), (t.x, t.y)))
            passed_triangles.append(triangle)

        self.plot(False)
        for edge in edges:
            plt.plot([edge[0][0], edge[1][0]], [edge[0][1], edge[1][1]], c='green')

        plt.show()

    def plot(self, show=True) -> None:
        plt.cla()
        plt.scatter([v[0] for v in self.vertexes], [v[1] for v in self.vertexes], c='r')

        ll = self.super_triangle.get_vertexes()
        for idx in range(len(ll)):
            p1 = ll[idx]
            p2 = ll[idx-1]
            plt.plot([p1[0], p2[0]], [p1[1], p2[1]], c='black')

        for triangle in self.triangles:
            ll = triangle.get_vertexes()
            for idx in range(len(ll)):
                p1 = ll[idx]
                p2 = ll[idx-1]
                plt.plot([p1[0], p2[0]], [p1[1], p2[1]], c='blue')
        if show:
            plt.show()
